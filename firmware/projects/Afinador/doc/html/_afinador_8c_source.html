163static void compararMagnitud()
164{
165 float frecMax = vector_frec[0];
166 float magnitudMax = audio_fft[0];
167 for (int i = 0; i < BUFFER_SIZE / 2; i++)
168 {
169 if (magnitudMax < audio_fft[i])
170 {
171 frecMax = vector_frec[i];
172 magnitudMax = audio_fft[i];
173 }
174 }
175
176 if (frecMax < (frecPrimeraCuerda - tolerancia))
177 {
178 enviarDatos(0);
179 }
180 else if (frecMax > (frecPrimeraCuerda - tolerancia) && frecMax < (frecPrimeraCuerda + tolerancia))
181 {
182 enviarDatos(1);
183 }
184 else if (frecMax > (frecPrimeraCuerda + tolerancia))
185 {
186 enviarDatos(2);
187 }
188}

195static void analizarAudio()
196{
197 while (1)
198 {
199 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
200 LowPassFilter(audio, audio_filt, BUFFER_SIZE);
201 HiPassFilter(audio_filt, audio_filt, BUFFER_SIZE);
202 FFTFrequency(SAMPLE_FREQ, BUFFER_SIZE, vector_frec);
203 FFTMagnitude(audio_filt, audio_fft, BUFFER_SIZE);
204 compararMagnitud();
205 contador = 0;
206 TimerStart(TIMER_A);
207 }
208}
